  def initialize(start, finals, transition_function)
    raise ArgumentError, 'Invalid transition function' unless transition_function.is_a?(TransitionFunction)
    @start = start 
    @finals = finals
    @transition_function = transition_function
    @alphabet = transition_function.values.map{|transitions| transitions.keys}.flatten.to_set
  end
