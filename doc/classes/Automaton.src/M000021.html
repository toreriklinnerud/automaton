  def tag(name)
    tagged_start = start.tag(name)
    tagged_finals = finals.map{|state| state.tag(name)}.to_set
    tagged_transition_function = transition_function.key_value_map  do |state, transitions|
      tagged_transitions = transitions.value_map do |symbol, states|
        states.map{|s| s.tag(name)}.to_set
      end
      [state.tag(name),tagged_transitions]
    end
    new = self.class.new(tagged_start, tagged_finals, tagged_transition_function)
    raise "not returning a transition function" unless new.transition_function.is_a?(TransitionFunction)
    new
  end
