  def to_total(alphabet)
    raise ArgumentError unless alphabet.is_a?(Set)
    all_states = (reachable_states + [:x]).map{|state| [state, {}]}.to_hash.to_transition_function
    total_transition_function = all_states.merge(transition_function).value_map do |state, transitions|
      missing_symbols = (alphabet - transitions.keys.to_set)
      missing_transitions = missing_symbols.map{|symbol| [symbol, [:x]]}.to_hash
      transitions.merge(missing_transitions)
    end
    self.class.new(start, finals, total_transition_function).prune
  end
