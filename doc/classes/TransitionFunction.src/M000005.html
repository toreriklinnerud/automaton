  def merge(other)
    raise ArgumentError, 'Merging with something that is not a TransitionFunction' unless other.is_a?(TransitionFunction)
    new = {}
    self.each do |state, transitions|
      new[state] = TransitionFunction.merge_transitions(other[state],transitions)
    end
    other.each do |state, transitions|
      new[state] = transitions unless new.has_key?(state)
    end
    new.to_transition_function
  end
